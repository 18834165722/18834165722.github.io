class Solution {
    public int distanceBetweenBusStops(int[] distance, int start, int destination) {
        int a=0,b=0,c;
        if(start>destination){
            c=start;
            start=destination;
            destination=c;
        }
        for(int i=0;i<distance.length;i++){
            if (start <= i && i < destination) {
                a += distance[i];
            } else {
                b += distance[i];
            }
        }
        return Math.min(a,b);
    }
}
